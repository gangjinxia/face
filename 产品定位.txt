### 模块1：眼鼻唇五官量化（直接追加类方法）
```python
def calc_eye_nose_lip(self):
    # 眼数据
    le_l = self.get_point(33)
    le_r = self.get_point(133)
    re_l = self.get_point(362)
    re_r = self.get_point(263)
    left_eye_h = abs(self.get_point(145)[1] - self.get_point(159)[1])
    right_eye_h = abs(self.get_point(374)[1] - self.get_point(386)[1])
    inter_eye_dist = abs(re_l[0] - le_r[0])
    eye_avg_w = (np.linalg.norm(le_r-le_l) + np.linalg.norm(re_r-re_l)) / 2

    # 鼻子
    nose_left = self.get_point(234)
    nose_right = self.get_point(454)
    nose_width = np.linalg.norm(nose_right - nose_left)
    nose_len = abs(self.get_point(1)[1] - self.get_point(4)[1])

    # 嘴唇
    lip_left = self.get_point(61)
    lip_right = self.get_point(291)
    lip_w = np.linalg.norm(lip_right - lip_left)
    upper_lip_h = abs(self.get_point(0)[1] - self.get_point(13)[1])
    lower_lip_h = abs(self.get_point(17)[1] - self.get_point(14)[1])
    philtrum = abs(self.get_point(1)[1] - self.get_point(0)[1])

    return {
        "平均眼宽": round(float(eye_avg_w/self.w),3),
        "眼间距": round(float(inter_eye_dist/self.w),3),
        "单眼平均高度": round(float((left_eye_h+right_eye_h)/2/self.h),3),
        "鼻翼宽度占脸比": round(float(nose_width/self.w),3),
        "鼻长占脸比": round(float(nose_len/self.h),3),
        "嘴唇宽度占脸比": round(float(lip_w/self.w),3),
        "上唇厚度": round(float(upper_lip_h/self.h),3),
        "下唇厚度": round(float(lower_lip_h/self.h),3),
        "人中长度": round(float(philtrum/self.h),3)
    }
```
