import cv2
import numpy as np

def face_analysis(img: np.ndarray) -> tuple[np.ndarray, list]:
    """
    人脸分析主函数
    :param img: 输入图像 ndarray (BGR格式)
    :return: 绘制人脸框后的图像ndarray, 裁剪后的人脸ndarray列表
    """
    # 1. 加载人脸检测器（OpenCV内置Haar分类器）
    face_cascade = cv2.CascadeClassifier(
        cv2.data.haarcascades + "haarcascade_frontalface_default.xml"
    )

    # 2. ndarray图像处理：转灰度图（依旧是ndarray）
    gray: np.ndarray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)

    # 3. 人脸检测，返回人脸坐标 [(x,y,w,h)]
    faces = face_cascade.detectMultiScale(
        gray,
        scaleFactor=1.1,
        minNeighbors=5,
        minSize=(30, 30)
    )

    # 4. 在原图ndarray上绘制人脸框
    img_draw = img.copy()  # 复制ndarray，不修改原图
    face_crops = []  # 存储裁剪后的人脸ndarray
    for (x, y, w, h) in faces:
        # 绘制矩形框
        cv2.rectangle(img_draw, (x, y), (x + w, y + h), (0, 255, 0), 2)
        # 裁剪人脸 ROI：切片ndarray
        face_roi: np.ndarray = img[y:y+h, x:x+w]
        face_crops.append(face_roi)

    return img_draw, face_crops
